\begin{exercise}
  The matrix
  \[
    \begin{bmatrix}
      -3 & 1 & 2 & 8\\
      2 & 0 & 3 & 4\\
      1 & 2 & 7 & -4\\
      5 & -1 & 2 & 0
    \end{bmatrix}
  \]
  \begin{multipleChoice}
    \choice{is singular.}
    \choice[correct]{is nonsingular.}
  \end{multipleChoice}

  \begin{feedback}[correct]
    The matrix row-reduces to
    \[
      \begin{bmatrix}
        \leading{1}&  0 &  0 &  0\\
        0 & \leading{1} &  0 &  0\\
        0 &  0 & \leading{1} &  0\\
        0 &  0 &  0 & \leading{1}
      \end{bmatrix}
    \]
    which is the $4\times 4$ identity matrix, so the original matrix must be nonsingular.
  \end{feedback}
\end{exercise}
