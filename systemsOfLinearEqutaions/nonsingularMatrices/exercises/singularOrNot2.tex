\begin{exercise}
  The matrix
  \[
    \begin{bmatrix}
      2 & 3 & 1 & 4\\
      1 & 1 & 1 & 0\\
      -1 & 2 & 3 & 5\\
      1 & 2 & 1 & 3
    \end{bmatrix}
  \]
  \begin{multipleChoice}
    \choice[correct]{is singular.}
    \choice{is nonsingular.}
  \end{multipleChoice}

  \begin{feedback}[correct]
    The matrix row-reduces to
    \[
      \begin{bmatrix}
        \leading{1} & 0 & 0 & -2\\
        0 & \leading{1} & 0 & 3\\
        0 & 0 & \leading{1} & -1\\
        0 & 0 & 0 & 0
      \end{bmatrix}
    \]
    Since this is not the $4\times 4$ identity matrix, the matrix is singular.
  \end{feedback}
\end{exercise}
