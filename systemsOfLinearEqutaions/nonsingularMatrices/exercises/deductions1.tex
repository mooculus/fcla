\begin{exercise}
  A certain system of linear equations consists of 6 equations in 6
  variables.  Its coefficient matrix, $A$, is singular.

  Will the matrix $A$ row-reduce to the identity matrix?

  \begin{multipleChoice}
    \choice[correct]{No.}
    \choice{Yes.}
    \choice{It depends---we need more information about $A$.}
  \end{multipleChoice}

  What could be true about this system's solution set?  Select all that
  are possible.
  \begin{selectAll}
    \choice[correct]{There are no solutions.}
    \choice{There is exactly one solution.}
    \choice[correct]{There are infinitely many solutions.}
  \end{selectAll}
\end{exercise}
