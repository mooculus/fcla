\begin{exercise}
  Suppose
  \[
    A = \begin{bmatrix} 1 & 3 & 1 \\ 1 & 2 & 1 \\ 2 & 2 & 1\end{bmatrix}
  \]
  Then the matrix $A$ is \wordChoice{\choice[correct]{invertible}\choice{not invertible}} and
  \[
    A^{-1} =  \begin{bmatrix} \answer{0} & -1 & 1\\ \answer{1} & -1 & 0\\\answer{-2} & 4 & -1 \end{bmatrix}.
  \]
\end{exercise}
